\section{Algorithms}
\subsection{Pseudocode}
\begin{frame}{Pseudocode}
  \begin{algorithm}[H]
    \caption{Greedy Selection (toy)}
    \begin{algorithmic}[1]
      \Require set $S$, score $w(\cdot)$, budget $k$
      \Ensure  subset $A$
      \State $A \gets \emptyset$
      \While{$|A| < k$}
        \State $x \gets \arg\max_{y\in S\setminus A} w(y)$
        \State $A \gets A \cup \{x\}$
      \EndWhile
      \State \Return $A$
    \end{algorithmic}
  \end{algorithm}
\end{frame}
